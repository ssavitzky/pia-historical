/etc/pcmcia/fixed.opts
uncomment DO_MOUNT and MOUNTPT add RUN_COMMAND="y" variable and COMMAND="~pia/pia/bin/copy_j6i $MOUNTPT ~pia/.pia/Photo_Album".
/etc/pcmcia/fixed
add 
  if [ !  $FAIL && "$RUN_COMMAND" = "y" ] ; then
          $COMMAND || FAIL=1
       fi
